CREATE TABLE IF NOT EXISTS customers (
    id INT AUTO_INCREMENT PRIMARY KEY,
    email VARCHAR(100) UNIQUE NOT NULL,
    first_name VARCHAR(100) NULL,
    last_name VARCHAR(100) NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE IF NOT EXISTS orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    customer_id INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    total_price DECIMAL(10,2),	
    currency CHAR(3),
    financial_status TINYINT, 
    -- 0 pending 
    -- 1 authorized
    -- 2 paid
    -- 3 refunded
    fulfillment_status TINYINT NOT NULL DEFAULT 0, 
    -- 0 = unfulfilled
    -- 1 = partial
    -- 2 = fulfilled
    -- 3 = shipped
    -- 4 = delivered
    -- 5 = returned

    FOREIGN KEY (customer_id) REFERENCES customers(id) ON DELETE CASCADE
);

CREATE TABLE user_profiles (
    user_id INT PRIMARY KEY,
    about_me TEXT NULL,
    favorite_quotes TEXT NULL,
    location VARCHAR(100) NULL,
    website VARCHAR(255) NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);


CREATE TABLE articles (
    id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    user_id INT NULL,
    literature INT NOT NULL, -- value 1 to 7 only
    series_id INT NULL,
    featured TINYINT NOT NULL,
    best TINYINT NULL,
    best_rankings INT NULL, 
    image_view VARCHAR(100) NULL, 	
    image_thumbnail VARCHAR(100) NULL,
    pictureAltText VARCHAR(200) NULL,  	
    headline VARCHAR(200) NOT NULL,
    summaryLead VARCHAR(300) NOT NULL,
    body LONGTEXT NOT NULL,
    updated_by_user_id INT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    is_deleted BOOLEAN DEFAULT FALSE,
    deleted_at TIMESTAMP NULL,         
    deleted_by INT NULL,
    FULLTEXT(headline, summaryLead, body),
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE SET NULL,
    FOREIGN KEY (updated_by_user_id) REFERENCES users(id) ON DELETE SET NULL,
    FOREIGN KEY (deleted_by) REFERENCES users(id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;




CREATE TABLE comments (
    id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    article_id INT NOT NULL,
    user_id INT NOT NULL,
    comment TEXT NOT NULL,
    status TINYINT NOT NULL DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX (article_id, created_at),  
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (article_id) REFERENCES articles(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;



CREATE TABLE ratings_articles (
    id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    article_id INT NOT NULL,
    user_id INT NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (article_id) REFERENCES articles(id) ON DELETE CASCADE,
    UNIQUE (article_id, user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;



CREATE TABLE invitation_codes ( 
    id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    referrer_user_id INT NOT NULL,  -- User who created this invitation code
    invitation_code VARCHAR(40) NOT NULL UNIQUE,  -- Unique code
    is_active TINYINT(1) DEFAULT 1,  -- Active by default
    deactivated_by_user_id INT DEFAULT NULL,  -- User who deactivates the code
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    expires_at DATETIME DEFAULT NULL,
    is_deleted BOOLEAN DEFAULT FALSE,  -- Soft delete flag
    FOREIGN KEY (referrer_user_id) REFERENCES users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;

CREATE TABLE activation_codes ( 
    id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    new_user_id INT NOT NULL,  
    activation_code VARCHAR(40) NOT NULL UNIQUE,  -- Unique code
    is_active TINYINT(1) DEFAULT 1,  
    deactivated_by_user_id INT DEFAULT NULL,  
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    expires_at DATETIME DEFAULT NULL,
    is_deleted BOOLEAN DEFAULT FALSE,  -- Soft delete flag
    FOREIGN KEY (new_user_id) REFERENCES users(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;







CREATE TABLE failed_actions (
  id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
  action_type ENUM('login', 'registration','activation') NOT NULL,
  username_attempted VARCHAR(255) NOT NULL,
  ip_address VARCHAR(45) NOT NULL,
  anti_bruteForce_token VARCHAR(255) NOT NULL,
  user_agent VARCHAR(512),
  fail_count SMALLINT UNSIGNED DEFAULT 1 NOT NULL,
  first_attempt DATETIME DEFAULT CURRENT_TIMESTAMP NOT NULL,
  last_attempt TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  UNIQUE (action_type, ip_address, anti_bruteForce_token)
)ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;








-- 1 is notifyAuthorComment, 2 is notifyReplyToComment
CREATE TABLE notificationComments (
  id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  type INT NOT NULL, 		
  recieverUserID INT NOT NULL,
  recieverArticleID INT NOT NULL,
  commentID INT NOT NULL,
  commentorName VARCHAR(100) NULL,
  followed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UNIQUE (commentID, recieverArticleID),
  FOREIGN KEY (recieverUserID) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (recieverArticleID) REFERENCES articles(id) ON DELETE CASCADE,
  FOREIGN KEY (commentID) REFERENCES comments(id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;















CREATE TABLE karmaLog (
    id BIGINT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    giverUserID INT NOT NULL,
    receiverUserID INT NOT NULL,
    contextType ENUM('article', 'comment') NOT NULL,
    contextID INT NOT NULL,
    given_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE (giverUserID, contextType, contextID),
    FOREIGN KEY (giverUserID) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (receiverUserID) REFERENCES users(id) ON DELETE CASCADE,
    CHECK (giverUserID <> receiverUserID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;
